import os
import numpy as np


DIR = "../../mmtime/data/airfoils"
airfoil = "flat0001"

fname = os.path.join(DIR,airfoil, airfoil + ".pol")
fname = os.path.abspath(fname)
print("loading:", fname)
if os.path.isfile(fname):
    print("decoding")
    columns = [
        "alpha",
        "CL",
        "CD",
        "CDp",
        "CM",
        "xtr_upper",
        "xtr_lower"
    ]
    try:
        data = np.genfromtxt(
            fname,
            skip_header=12,
            names=columns,
        )
    except OSError:
        data = np.array([])
print(data["CL"])
