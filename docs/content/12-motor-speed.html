def torque(n, Qm,x1,y1,x2,y2,Nm):
    M = (y2 - y1)/(x2 - x1)
    if n<x1: 
        A = (M*x1 + Qm - y1)/x1**2
        B = (-M*x1 - 2*Qm + 2*y1)/x1
        C = Qm
        return A*n**2 + B*n + C
    elif n < x2:
        M = (y2 - y1)/(x2 - x1)
        return y1 + M*(n - x1)
    else:
        xp = Nm-n
        A = (-M*Nm + M*x2 - y2)/(Nm**2 - 2*Nm*x2 + x2**2)
        B = (-M*Nm + M*x2 - 2*y2)/(-Nm + x2)
        C = 0
        return A*xp**2 + B*xp + C

def max_turns(length, weight):
    K = 8.48597 * u.ounce**0.5/u.inch**1.5
    Nmax = K * length / (weight/length)**0.5
    return Nmax
